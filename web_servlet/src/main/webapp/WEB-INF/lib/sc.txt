create table contents(
cidx int unsigned not null auto_increment,
c_part enum('광고','협찬','콘텐츠','디자인') not null default '광고',
c_corp varchar(50) not null default '',
c_name varchar(50) not null default '',
c_email varchar(100) not null,
c_tel char(11) not null default '00000000000',
c_text text not null,
c_agree enum('Y','N') not null default 'Y',
c_date s not null default current_timestamp,
primary key(cidx)
);
